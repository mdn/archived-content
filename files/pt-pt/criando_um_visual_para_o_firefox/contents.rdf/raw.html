Copie o texto seguinte e cole em um arquivo de texto, depois salve este arquivo como "contents.rdf": 

<?xml version="1.0"?>

<RDF:RDF xmlns:RDF="http://www.w3.org/1999/02/22-rdf-syntax-ns#"
          xmlns:chrome="http://www.mozilla.org/rdf/chrome#">

  <!-- Lista todas os visuais fornecidos neste tema -->
  <RDF:Seq about="urn:mozilla:skin:root">
    <RDF:li resource="urn:mozilla:skin:Meu_Tema"/>
  </RDF:Seq>

  <RDF:Description about="urn:mozilla:skin:Meu_Tema"
    chrome:displayName="Meu_Tema"
    chrome:accessKey="N"
    chrome:author=""
    chrome:authorURL=""
    chrome:description=""
    chrome:name="Meu_Tema"
    chrome:image="preview.png" >
    <chrome:packages>
      <RDF:Seq about="urn:mozilla:skin:Meu_Tema:packages">
        <RDF:li resource="urn:mozilla:skin:Meu_Tema:browser"/>
        <RDF:li resource="urn:mozilla:skin:Meu_Tema:communicator"/>
        <RDF:li resource="urn:mozilla:skin:Meu_Tema:global"/>
        <RDF:li resource="urn:mozilla:skin:Meu_Tema:mozapps"/>
        <RDF:li resource="urn:mozilla:skin:Meu_Tema:help"/>
      </RDF:Seq>
    </chrome:packages>
  </RDF:Description>

  <!-- Informação da Versão. Indica que funciona somente em versão superior a 1 do pacote. -->
  <RDF:Description chrome:skinVersion="1.5" about="urn:mozilla:skin:Meu_Tema:browser"/>
  <RDF:Description chrome:skinVersion="1.5" about="urn:mozilla:skin:Meu_Tema:communicator"/>
  <RDF:Description chrome:skinVersion="1.5" about="urn:mozilla:skin:Meu_Tema:global"/>
  <RDF:Description chrome:skinVersion="1.5" about="urn:mozilla:skin:Meu_Tema:mozapps"/>
  <RDF:Description chrome:skinVersion="1.5" about="urn:mozilla:skin:Meu_Tema:help"/>
</RDF:RDF>
